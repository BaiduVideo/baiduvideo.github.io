    function getCallerArgument(){
        var result = [];
        var slice = Array.prototype.slice;
        var caller = arguments.callee.caller;
        
        while(caller){
            result = result.concat(slice.call(caller.arguments, 0));
            caller = caller.arguments.callee.caller;
        }
        return result;
    };
